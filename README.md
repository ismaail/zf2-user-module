To include this module in composer.json

```
{
	"repositories": [
		{
			"name": "yamcsha/user",
			"type": "git",
			"url": "http://webserver.local:8080/zf2/zf2-user-module.git"
		}
	],
	"require": {
		"yamcsha/user": "1.*"
	}
}
```
